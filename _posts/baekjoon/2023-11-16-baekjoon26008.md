import java.util.Scanner;

public class Main {
    static final int div = (int) (1e9 + 7);

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();
        int m = sc.nextInt();
        int a = sc.nextInt();
        int h = sc.nextInt();

        long answer = 1;
        for (int i = 0; i < n - 1; i++) {
            answer = (answer * m) % div;
        }
        System.out.println(answer);
    }
}
